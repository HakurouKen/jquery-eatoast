Jquery-eatoast

A simple jQuery based toast plugin

Download this project as a .zip file Download this project as a tar.gz file

Basic

Default 

$.toast.show("This is a default toast.");

Success 

$.toast.success("This is a success toast >_<");

Info 

$.toast.info("This is a info toast.");

Warning 

$.toast.warn("This is a warning toast...");

Danger 

$.toast.danger("This is a danger toast!!");

Custom

SlideIn 

$.toast.show({
	animate: 'slide',
	text: 'This text will slide in'
});

Close button 

$.toast.show({
	animate: 'slide',
	autoclose: false,
	closeBtn: true,
	style: 'warn',
	text: 'This text will not close until you click close button.'
});

Align & Position 

$.toast.show({
	align: 'bottom',
	position: '10%',
	text: 'This text will fix at bottom.'
});

Others 

$.toast.show({
	duration: 5000,
	color: '#DDD',
	background: '#333',
	border: 'none',
	opacity: '1',
	speed: 'slow',
	text: 'Many other options...'
});
